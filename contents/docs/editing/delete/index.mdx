## Delete

Remove rows from the table programmatically or through context menus.

### Programmatic Delete

```tsx
const handleDeleteRow = (index: number) => {
  const newItems = items.filter((_, i) => i !== index)
  setItems(newItems)

  // Delete from database
  deleteItem(items[index].id)
}
```

### Example with Confirmation

```tsx
const handleDeleteRow = async (index: number) => {
  const confirmed = window.confirm("Delete this row?")

  if (confirmed) {
    try {
      await deleteItem(items[index].id)

      const newItems = items.filter((_, i) => i !== index)
      setItems(newItems)

      showSuccess("Row deleted")
    } catch (error) {
      showError("Failed to delete row")
    }
  }
}
```

### Bulk Delete

Delete multiple rows at once:

```tsx
const handleDeleteSelected = async (selectedIndices: number[]) => {
  const confirmed = window.confirm(`Delete ${selectedIndices.length} rows?`)

  if (confirmed) {
    try {
      // Delete from database
      const idsToDelete = selectedIndices.map((i) => items[i].id)
      await deleteBulk(idsToDelete)

      // Update local state
      const newItems = items.filter((_, i) => !selectedIndices.includes(i))
      setItems(newItems)

      showSuccess(`${selectedIndices.length} rows deleted`)
    } catch (error) {
      showError("Failed to delete rows")
    }
  }
}
```

### Undo Delete

Implement undo functionality:

```tsx
const [deletedRows, setDeletedRows] = useState([])

const handleDeleteRow = async (index: number) => {
  const deletedRow = items[index]

  // Store deleted row
  setDeletedRows([...deletedRows, { item: deletedRow, index }])

  // Remove from table
  const newItems = items.filter((_, i) => i !== index)
  setItems(newItems)

  // Delete from database
  deleteItem(deletedRow.id)
}

const handleUndo = () => {
  if (deletedRows.length > 0) {
    const { item, index } = deletedRows[deletedRows.length - 1]

    // Restore row
    const newItems = [...items]
    newItems.splice(index, 0, item)
    setItems(newItems)

    // Remove from undo stack
    setDeletedRows(deletedRows.slice(0, -1))
  }
}
```
