## Headers Context Menu

Right-click on column headers to perform column operations.

### Implementation

```tsx
const handleHeaderContextMenu = (
  e: React.MouseEvent,
  header: BaseTableHeader
) => {
  e.preventDefault()

  showContextMenu({
    x: e.clientX,
    y: e.clientY,
    items: [
      {
        label: "Sort ascending",
        onClick: () => sortColumn(header.id, "asc"),
      },
      {
        label: "Sort descending",
        onClick: () => sortColumn(header.id, "desc"),
      },
      { divider: true },
      {
        label: "Filter",
        onClick: () => openFilterDialog(header),
      },
      { label: "Hide column", onClick: () => hideColumn(header.id) },
      { divider: true },
      {
        label: "Column settings",
        onClick: () => openColumnSettings(header),
      },
    ],
  })
}
```

### Available Actions

| Action          | Description           |
| --------------- | --------------------- |
| Sort ascending  | A → Z or 0 → 9        |
| Sort descending | Z → A or 9 → 0        |
| Filter          | Open filter options   |
| Hide column     | Remove from view      |
| Show column     | Restore hidden column |
| Rename          | Change header text    |
| Resize          | Adjust width          |
| Freeze          | Pin to side           |
| Settings        | Column configuration  |

### Sort Column

```tsx
const handleSort = (columnId: string, direction: "asc" | "desc") => {
  const sorted = [...items].sort((a, b) => {
    if (a[columnId] < b[columnId]) return direction === "asc" ? -1 : 1
    if (a[columnId] > b[columnId]) return direction === "asc" ? 1 : -1
    return 0
  })

  setItems(sorted)
}
```

### Hide/Show Columns

```tsx
const [hiddenColumns, setHiddenColumns] = useState<string[]>([])

const headers = originalHeaders.filter((h) => !hiddenColumns.includes(h.id))

const handleHideColumn = (columnId: string) => {
  setHiddenColumns([...hiddenColumns, columnId])
}

const handleShowColumn = (columnId: string) => {
  setHiddenColumns(hiddenColumns.filter((id) => id !== columnId))
}
```
