## Highlight a Cell

Apply conditional styling to cells for visual organization.

### Basic Highlighting

```tsx
<BaseTable
  headers={headers}
  items={items}
  highlightCondition={[
    {
      propertyId: "status",
      value: "in progress",
      style: { backgroundColor: "#ffeb3b", color: "#000" },
      columnId: "status",
    },
  ]}
/>
```

### HighlightCondition Structure

```ts
import { type CSSProperties } from "react"

export default interface HighlightCondition {
  propertyId: string
  value: unknown
  style: CSSProperties
  columnId?: string // Optional, if provided, applies the style to the specific column only
}
```

### Set Highlight

Allow users to highlight cells:

```tsx
<BaseTable
  headers={headers}
  items={items}
  highlightCondition={highlights}
  onSetHighlightCondition={(highlight, item) => {
    // Add highlight
    const newHighlights = [...highlights, highlight]
    setHighlights(newHighlights)
    saveHighlight(highlight)
  }}
/>
```

### Remove Highlight

Remove highlights from cells:

```tsx
onRemoveHighlightCondition={(highlight, item) => {
  // Remove the highlight
  const newHighlights = highlights.filter(
    (h) => h.propertyId !== highlight.propertyId || h.value !== highlight.value
  )
  setHighlights(newHighlights)
}}
```

### Common Highlight Colors

```tsx
// Warning
{ backgroundColor: "#fff3cd", color: "#856404" }

// Error
{ backgroundColor: "#f8d7da", color: "#721c24" }

// Success
{ backgroundColor: "#d4edda", color: "#155724" }

// Info
{ backgroundColor: "#cce5ff", color: "#004085" }
```

### Example: Priority Highlighting

```tsx
const getPriorityColor = (priority: string) => {
  const colors: Record<string, CSSProperties> = {
    high: { backgroundColor: "#f8d7da", color: "#721c24" },
    medium: { backgroundColor: "#fff3cd", color: "#856404" },
    low: { backgroundColor: "#d4edda", color: "#155724" },
  }
  return colors[priority] || {}
}

// Apply highlights based on data
const highlights = [
  {
    propertyId: "priority",
    value: "high",
    style: getPriorityColor("high"),
    columnId: "priority",
  },
  {
    propertyId: "priority",
    value: "medium",
    style: getPriorityColor("medium"),
    columnId: "priority",
  },
]
```

### Persist Highlights

Save highlight state:

```tsx
onSetHighlightCondition={async (highlight, item) => {
  await saveHighlight(highlight)
  setHighlights([...highlights, highlight])
}}
```
